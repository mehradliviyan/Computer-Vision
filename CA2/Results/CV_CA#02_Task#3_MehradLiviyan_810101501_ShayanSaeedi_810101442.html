import cv2
import matplotlib.pyplot as plt
import os
import numpy as np

NUM_SAMPLES = 10
CLEANED_PATH = "../Results/Binary_Cleaned"

cleaned_images = []

for i in range(NUM_SAMPLES):
    image = cv2.imread(os.path.join(CLEANED_PATH, f"{i+1:02d}.png"), cv2.IMREAD_GRAYSCALE)

    if image is None:
        print(f"Error: Image {i+1:02d}.png not found or path is incorrect")
        continue

    cleaned_images.append(image)
    print(f"Image {i+1:02d}.png loaded successfully")

    plt.imshow(image, cmap="gray")
    plt.axis("off")
    plt.show()

SEGMENTED_PATH = "../Results/Segmented_Characters"
os.makedirs(SEGMENTED_PATH, exist_ok=True)

TARGET_SIZE = (64, 64)

for idx in range(NUM_SAMPLES):
    img = cleaned_images[idx] 
    img = cv2.bitwise_not(img)
    
    num_labels, labels, stats, _ = cv2.connectedComponentsWithStats(img, connectivity=8)

    print(f"Image {idx+1:02d} found {num_labels-1} chars")
    plt.figure(figsize=(6,4))
    plt.imshow(img, cmap="gray")
    plt.title(f"Original Cleaned Image {idx+1:02d}")
    plt.axis("off")
    plt.show()

    char_count = 0
    char_folder = os.path.join(SEGMENTED_PATH, f"{idx+1:02d}")
    os.makedirs(char_folder, exist_ok=True)

    for label in range(1, num_labels):  
        x, y, w, h, area = stats[label]

        char_img = img[y:y+h, x:x+w]
        char_resized = cv2.resize(char_img, TARGET_SIZE, interpolation=cv2.INTER_NEAREST)
        cv2.imwrite(os.path.join(char_folder, f"char_{char_count+1}.png"), char_resized)
        plt.figure(figsize=(2,2))
        plt.imshow(char_resized, cmap="gray")
        plt.title(f"Char {char_count+1}")
        plt.axis("off")
        plt.show()

        char_count += 1
